64@class PartitionerBase
65
66@brief class to derive a partitioner for scheduling parallel algorithms
67
68@tparam C closure wrapper type
69
70The class provides base methods to derive a partitioner that can be used
71to schedule parallel iterations (e.g., tf::Taskflow::for_each).
72
73An partitioner defines the scheduling method for running parallel algorithms,
74such tf::Taskflow::for_each, tf::Taskflow::reduce, and so on.
75By default, we provide the following partitioners: 
76
77+ tf::GuidedPartitioner  to enable guided scheduling algorithm of adaptive chunk size
78+ tf::DynamicPartitioner to enable dynamic scheduling algorithm of equal chunk size
79+ tf::StaticPartitioner  to enable static scheduling algorithm of static chunk size
80+ tf::RandomPartitioner  to enable random scheduling algorithm of random chunk size
81
82Depending on applications, partitioning algorithms can impact the performance
83a lot. 
84For example, if a parallel-iteration workload contains a regular work unit per
85iteration, tf::StaticPartitioner can deliver the best performance.
86On the other hand, if the work unit per iteration is irregular and unbalanced,
87tf::GuidedPartitioner or tf::DynamicPartitioner can outperform tf::StaticPartitioner.
88In most situations, tf::GuidedPartitioner can deliver decent performance and
89is thus used as our default partitioner.
90
91@attention
92Giving the partition size of 0 lets the %Taskflow runtime automatically determines
93the partition size for the given partitioner.
94
95
96In addition to partition size, the application can specify a closure wrapper
97for a partitioner.
98A closure wrapper allows the application to wrap a partitioned task 
99(i.e., closure) with a custom function object that performs additional tasks.
100For example:
101
102@code{.cpp}
103std::atomic<int> count = 0;
104tf::Taskflow taskflow;
105taskflow.for_each_index(0, 100, 1, 
106  [](){                 
107    printf("%d\n", i); 
108  },
109  tf::StaticPartitioner(0, [](auto&& closure){
110    // do something before invoking the partitioned task
111    // ...
112    
113    // invoke the partitioned task
114    closure();
115
116    // do something else after invoking the partitioned task
117    // ...
118  }
119);
120executor.run(taskflow).wait();
121@endcode
122
123@attention
124The default closure wrapper (tf::DefaultClosureWrapper) does nothing but invoke
125the partitioned task (closure).
126

644@brief class to construct a random partitioner for scheduling parallel algorithms
645
646@tparam C closure wrapper type (default tf::DefaultClosureWrapper)
647
648Similar to tf::DynamicPartitioner, 
649the partitioner splits iterations into many partitions but each with a random
650chunk size in the range, <tt>c = [alpha * N * W, beta * N * W]</tt>.
651By default, @c alpha is <tt>0.01</tt> and @c beta is <tt>0.5</tt>, respectively.
652
653In addition to partition size, the application can specify a closure wrapper
654for a random partitioner.
655A closure wrapper allows the application to wrap a partitioned task 
656(i.e., closure) with a custom function object that performs additional tasks.
657For example:
658
659@code{.cpp}
660std::atomic<int> count = 0;
661tf::Taskflow taskflow;
662taskflow.for_each_index(0, 100, 1, 
663  [](){                 
664    printf("%d\n", i); 
665  },
666  tf::RandomPartitioner(0, [](auto&& closure){
667    // do something before invoking the partitioned task
668    // ...
669    
670    // invoke the partitioned task
671    closure();
672
673    // do something else after invoking the partitioned task
674    // ...
675  }
676);
677executor.run(taskflow).wait();
678@endcode
679*/
680template <typename C = DefaultClosureWrapper>
681class RandomPartitioner : public PartitionerBase<C> {
682
683  public:
684  
685  /**
686  @brief queries the partition type (dynamic)
687  */
688  static constexpr PartitionerType type() { return PartitionerType::DYNAMIC; }
689
690  /**
691  @brief default constructor
692  */
693  RandomPartitioner() = default;
694  
695  /**
696  @brief construct a dynamic partitioner with the given chunk size
697  */
698  explicit RandomPartitioner(size_t sz) : PartitionerBase<C>(sz) {}
699  
700  /**
701  @brief construct a random partitioner with the given chunk size and the closure
702  */ 
703  explicit RandomPartitioner(size_t sz, C&& closure) :
704    PartitionerBase<C>(sz, std::forward<C>(closure)) {
705  }
706  
707  /**
708  @brief constructs a random partitioner with the given parameters
709  */
710  RandomPartitioner(float alpha, float beta) : _alpha{alpha}, _beta{beta} {}
711
712  /**
713  @brief constructs a random partitioner with the given parameters and the closure
714  */
715  RandomPartitioner(float alpha, float beta, C&& closure) : 
716    _alpha {alpha}, _beta {beta}, 
717    PartitionerBase<C>(0, std::forward<C>(closure)) {
718  }
719
720  /**
721  @brief queries the @c alpha value
722  */
723  float alpha() const { return _alpha; }
724  
725  /**
726  @brief queries the @c beta value
727  */
728  float beta() const { return _beta; }
729  
730  /**
731  @brief queries the range of chunk size
732  
733  @param N number of iterations
734  @param W number of workers
735  */
736  std::pair<size_t, size_t> chunk_size_range(size_t N, size_t W) const {
737    
738    size_t b1 = static_cast<size_t>(_alpha * N * W);
739    size_t b2 = static_cast<size_t>(_beta  * N * W);
740
741    if(b1 > b2) {
742      std::swap(b1, b2);
743    }
744
745    b1 = (std::max)(b1, size_t{1});
746    b2 = (std::max)(b2, b1 + 1);
747
748    return {b1, b2};
749  }
750
751  // --------------------------------------------------------------------------
752  // scheduling methods
753  // --------------------------------------------------------------------------
754  
755  /**
756  @private
757  */
758  template <typename F, 
759    std::enable_if_t<std::is_invocable_r_v<void, F, size_t, size_t>, void>* = nullptr
760  >
761  void loop(
762    size_t N, size_t W, std::atomic<size_t>& next, F&& func
763  ) const {
764
765    auto [b1, b2] = chunk_size_range(N, W); 
766    
767    std::default_random_engine engine {std::random_device{}()};
768    std::uniform_int_distribution<size_t> dist(b1, b2);
769    
770    size_t chunk_size = dist(engine);
771    size_t curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
772
773    while(curr_b < N) {
774      func(curr_b, (std::min)(curr_b + chunk_size, N));
775      chunk_size = dist(engine);
776      curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
777    }
778  }
779
780  /**
781  @private
782  */
783  template <typename F, 
784    std::enable_if_t<std::is_invocable_r_v<bool, F, size_t, size_t>, void>* = nullptr
785  >
786  void loop_until(
787    size_t N, size_t W, std::atomic<size_t>& next, F&& func
788  ) const {
789
790    auto [b1, b2] = chunk_size_range(N, W); 
791    
792    std::default_random_engine engine {std::random_device{}()};
793    std::uniform_int_distribution<size_t> dist(b1, b2);
794    
795    size_t chunk_size = dist(engine);
796    size_t curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
797
798    while(curr_b < N) {
799      if(func(curr_b, (std::min)(curr_b + chunk_size, N))){
800        return;
801      }
802      chunk_size = dist(engine);
803      curr_b = next.fetch_add(chunk_size, std::memory_order_relaxed);
804    }
805  }
806
807  private:
808
809  float _alpha {0.01f};
810  float _beta  {0.50f};
811};
