313  template <typename T>
314  Tensor<T> Tensor<T>::broadcast_to(const std::vector<size_t>& target_shape) const
315  {
316    size_t ndim = target_shape.size();
317    size_t src_ndim = m_shape.size();
318
319    if (ndim < src_ndim)
320    {
321      throw std::invalid_argument("Cannot broadcast shape " +
322          shape_to_string(m_shape) + " to " + shape_to_string(target_shape) +
323          ": target has fewer dimensions");
324    }
325
326    Tensor ret;
327    ret.m_data = m_data;
328    ret.m_offset = m_offset;
329    ret.m_shape = target_shape;
330    ret.m_strides.resize(ndim);
331    ret.m_isContiguous = false;
332
333    size_t prepended = ndim - src_ndim;
334    for (size_t i = 0; i < ndim; ++i)
335    {
336      if (i < prepended)
337      {
338        ret.m_strides[i] = 0;
339      }
340      else
341      {
342        size_t si = i - prepended;
343        if (m_shape[si] == target_shape[i])
344        {
345          ret.m_strides[i] = m_strides[si];
346        }
347        else if (m_shape[si] == 1)
348        {
349          ret.m_strides[i] = 0;
350        }
351        else
352        {
353          throw std::invalid_argument("Cannot broadcast shape " +
354              shape_to_string(m_shape) + " to " + shape_to_string(target_shape));
355        }
356      }
357    }
358
359    return ret;
360  }

366    /// Apply a binary operation elementwise to two broadcast-compatible tensors.
367    /// The result element type R is deduced from the return type of op.
368    template <typename T, typename BinaryOp,
369              typename R = std::invoke_result_t<BinaryOp, const T&, const T&>>
370    Tensor<R> broadcast_binop(const Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
371    {
372      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
373      auto lhs_view = lhs.broadcast_to(out_shape);
374      auto rhs_view = rhs.broadcast_to(out_shape);
375      Tensor<R> result(out_shape);
376      result.walk([&](const std::vector<size_t>& idx) {
377        result(idx) = op(lhs_view(idx), rhs_view(idx));
378      });
379      return result;
380    }
381
382    template <typename T, typename BinaryOp>
383    Tensor<T>& broadcast_binop_inplace(Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
384    {
385      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
386      if (out_shape != lhs.shape())
387        throw std::invalid_argument("Cannot broadcast in-place: output shape " +
388            shape_to_string(out_shape) + " differs from LHS shape " + shape_to_string(lhs.shape()));
389      auto rhs_view = rhs.broadcast_to(out_shape);
390      lhs.walk([&](const std::vector<size_t>& idx) {
391        lhs(idx) = op(lhs(idx), rhs_view(idx));
392      });
393      return lhs;
394    }

465  template <typename T>
466  Tensor<T> masked_select(const Tensor<T>& src, const Tensor<bool>& mask)
467  {
468    if (mask.shape() != src.shape())
469    {
470      throw std::invalid_argument(
471        "masked_select: mask shape does not match tensor shape");
472    }
473
474    // Single pass: collect matching elements into a buffer
475    std::vector<T> buf;
476    walk(src, [&](const std::vector<size_t>& idx) {
477      if (mask(idx))
478        buf.push_back(src(idx));
479    });
480
481    if (buf.empty())
482    {
483      return Tensor<T>({0});
484    }
485
486    Tensor<T> result({buf.size()});
487    std::move(buf.begin(), buf.end(), result.data());
488
489    return result;
490  }

492  template <typename T>
493  void masked_assign(Tensor<T>& dst, const Tensor<bool>& mask, const Tensor<T>& values)
494  {
495    if (mask.shape() != dst.shape())
496    {
497      throw std::invalid_argument(
498        "masked_assign: mask shape does not match tensor shape");
499    }
500
501    size_t pos = 0;
502    walk(dst, [&](const std::vector<size_t>& idx) {
503      if (mask(idx))
504      {
505        if (pos >= values.size())
506        {
507          throw std::invalid_argument(
508            "masked_assign: more true values in mask than elements in values");
509        }
510        dst(idx) = values({pos++});
511      }
512    });
513
514    if (pos != values.size())
515    {
516      throw std::invalid_argument(
517        "masked_assign: values length (" + std::to_string(values.size()) +
518        ") does not match number of true values in mask (" + std::to_string(pos) + ")");

567  Tensor<T> axis_select(const Tensor<T>& src, size_t axis, const Tensor<bool>& mask)
568  {
569    detail::validate_axis_mask(axis, src.shape().size(), mask, src.shape()[axis]);
570    auto true_indices = detail::collect_true_indices(mask);
571
572    // Build output shape
573    auto out_shape = src.shape();
574    out_shape[axis] = true_indices.size();
575
576    if (true_indices.empty())
577    {
578      return Tensor<T>(out_shape);

580
581    Tensor<T> result(out_shape);
582    walk(result, [&](const std::vector<size_t>& out_idx) {
583      auto src_idx = out_idx;
584      src_idx[axis] = true_indices[out_idx[axis]];
585      result(out_idx) = src(src_idx);

708    struct ResolvedSelector
709    {
710      size_t axis;
711      std::vector<size_t> indices;  // selected positions along this axis
712    };
713
714    inline std::vector<ResolvedSelector> resolve_selectors(
715      const std::vector<std::pair<size_t, AxisSelector>>& selectors,
716      const std::vector<size_t>& shape)
717    {
718      std::vector<ResolvedSelector> result;
719      result.reserve(selectors.size());
720      for (const auto& [axis, sel] : selectors)
721      {
722        if (axis >= shape.size())
723          throw std::invalid_argument("outer indexing: axis " + std::to_string(axis) +
724            " out of range for tensor with " + std::to_string(shape.size()) + " dimensions");
725
726        ResolvedSelector rs;
727        rs.axis = axis;
728        std::visit([&](const auto& idx_tensor) {
729          using IdxT = std::decay_t<decltype(idx_tensor)>;
730          if (idx_tensor.shape().size() != 1)
731            throw std::invalid_argument("outer indexing: selector must be 1D");
732
733          if constexpr (std::is_same_v<IdxT, Tensor<bool>>)
734          {
735            if (idx_tensor.shape()[0] != shape[axis])
736              throw std::invalid_argument("outer indexing: bool mask length must match axis size");
737            rs.indices = collect_true_indices(idx_tensor);
738          }
739          else
740          {
741            rs.indices.reserve(idx_tensor.shape()[0]);
742            for (size_t i = 0; i < idx_tensor.shape()[0]; ++i)
743              rs.indices.push_back(static_cast<size_t>(idx_tensor({i})));
744          }
745        }, sel);
746        result.push_back(std::move(rs));
747      }
748      return result;
749    }
750  }

752  template <typename T>
753  Tensor<T> outer_select(const Tensor<T>& src,
754    const std::vector<std::pair<size_t, AxisSelector>>& selectors)
755  {
756    auto resolved = detail::resolve_selectors(selectors, src.shape());
757
758    auto out_shape = src.shape();
759    for (const auto& rs : resolved)
760      out_shape[rs.axis] = rs.indices.size();
761
762    Tensor<T> result(out_shape);
763    walk(result, [&](const std::vector<size_t>& out_idx) {
764      auto src_idx = out_idx;
765      for (const auto& rs : resolved)
766        src_idx[rs.axis] = rs.indices[out_idx[rs.axis]];
767      result(out_idx) = src(src_idx);
768    });
769
770    return result;
771  }

773  template <typename T>
774  void outer_fill(Tensor<T>& dst,
775    const std::vector<std::pair<size_t, AxisSelector>>& selectors,
776    const T& value)
777  {
778    auto resolved = detail::resolve_selectors(selectors, dst.shape());
779
780    // Build a fast lookup: for each axis with a selector, a set of selected indices
781    std::vector<std::vector<bool>> axis_selected(dst.shape().size());
782    for (const auto& rs : resolved)
783    {
784      axis_selected[rs.axis].assign(dst.shape()[rs.axis], false);
785      for (size_t idx : rs.indices)
786        axis_selected[rs.axis][idx] = true;
787    }
788
789    walk(dst, [&](const std::vector<size_t>& idx) {
790      for (const auto& rs : resolved)
791      {
792        if (!axis_selected[rs.axis][idx[rs.axis]])
793          return;
794      }
795      dst(idx) = value;
796    });
797  }

844  template <typename T>
845  Tensor<T> concatenate(const std::vector<Tensor<T>>& tensors, size_t axis)
846  {
847    if (tensors.empty())
848      throw std::invalid_argument("concatenate: need at least one tensor");
849
850    auto ndim = tensors[0].shape().size();
851    if (axis >= ndim)
852      throw std::invalid_argument("concatenate: axis " + std::to_string(axis) +
853        " out of range for " + std::to_string(ndim) + "-D tensor");
854
855    // Validate shapes match on all axes except the join axis
856    for (size_t t = 1; t < tensors.size(); ++t)
857    {
858      if (tensors[t].shape().size() != ndim)
859        throw std::invalid_argument("concatenate: all tensors must have the same number of dimensions");
860      for (size_t d = 0; d < ndim; ++d)
861      {
862        if (d != axis && tensors[t].shape()[d] != tensors[0].shape()[d])
863          throw std::invalid_argument("concatenate: shape mismatch on dimension " + std::to_string(d));
864      }
865    }
866
867    // Compute output shape
868    auto out_shape = tensors[0].shape();
869    out_shape[axis] = 0;
870    for (const auto& t : tensors)
871      out_shape[axis] += t.shape()[axis];
872
873    Tensor<T> result(out_shape);
874
875    // Copy each tensor's data into the result
876    size_t offset = 0;
877    for (const auto& src : tensors)
878    {
879      auto src_axis_size = src.shape()[axis];
880      walk(src, [&](const std::vector<size_t>& src_idx) {
881        auto dst_idx = src_idx;
882        dst_idx[axis] += offset;
883        result(dst_idx) = src(src_idx);
884      });
885      offset += src_axis_size;
886    }
887
888    return result;
889  }

891  template <typename T>
892  Tensor<T> stack(const std::vector<Tensor<T>>& tensors, ptrdiff_t axis)
893  {
894    if (tensors.empty())
895      throw std::invalid_argument("stack: need at least one tensor");
896
897    auto ndim = static_cast<ptrdiff_t>(tensors[0].shape().size());
898
899    // Resolve negative axis; valid range for new axis is [-(ndim+1), ndim]
900    if (axis < 0)
901      axis += ndim + 1;
902    if (axis < 0 || axis > ndim)
903      throw std::invalid_argument("stack: axis out of range");
904
905    // Validate all shapes match
906    for (size_t t = 1; t < tensors.size(); ++t)
907    {
908      if (tensors[t].shape() != tensors[0].shape())
909        throw std::invalid_argument("stack: all tensors must have the same shape");
910    }
911
912    // expand_dims each tensor, then concatenate along the new axis
913    std::vector<Tensor<T>> expanded;
914    expanded.reserve(tensors.size());
915    for (const auto& t : tensors)
916      expanded.push_back(t.expand_dims(axis));
917
918    return concatenate(expanded, static_cast<size_t>(axis));
919  }

921  template <typename T>
922  std::vector<Tensor<T>> split(const Tensor<T>& tensor,
923    const std::vector<size_t>& split_points, size_t axis)
924  {
925    auto ndim = tensor.shape().size();
926    if (axis >= ndim)
927      throw std::invalid_argument("split: axis " + std::to_string(axis) +
928        " out of range for " + std::to_string(ndim) + "-D tensor");
929
930    auto axis_size = tensor.shape()[axis];
931
932    // Build slice boundaries: [0, sp0, sp1, ..., axis_size]
933    std::vector<size_t> boundaries;
934    boundaries.reserve(split_points.size() + 2);
935    boundaries.push_back(0);
936    for (auto sp : split_points)
937    {
938      if (sp > axis_size)
939        sp = axis_size;
940      boundaries.push_back(sp);
941    }
942    boundaries.push_back(axis_size);
943
944    std::vector<Tensor<T>> result;
945    result.reserve(boundaries.size() - 1);
946    for (size_t i = 0; i + 1 < boundaries.size(); ++i)
947    {
948      std::vector<Slice> slices(ndim, all());
949      slices[axis] = range(
950        static_cast<ptrdiff_t>(boundaries[i]),
951        static_cast<ptrdiff_t>(boundaries[i + 1]),
952        std::nullopt);
953      result.push_back(tensor[slices]);
954    }
955    return result;
956  }

958  template <typename T>
959  std::vector<Tensor<T>> split(const Tensor<T>& tensor,
960    size_t n_sections, size_t axis)
961  {
962    auto ndim = tensor.shape().size();
963    if (axis >= ndim)
964      throw std::invalid_argument("split: axis " + std::to_string(axis) +
965        " out of range for " + std::to_string(ndim) + "-D tensor");
966
967    auto axis_size = tensor.shape()[axis];
968    if (axis_size % n_sections != 0)
969      throw std::invalid_argument("split: tensor of size " +
970        std::to_string(axis_size) + " along axis " + std::to_string(axis) +
971        " cannot be evenly split into " + std::to_string(n_sections) + " sections");
972
973    auto section_size = axis_size / n_sections;
974    std::vector<size_t> split_points;
975    split_points.reserve(n_sections - 1);
976    for (size_t i = 1; i < n_sections; ++i)
977      split_points.push_back(i * section_size);
978
979    return split(tensor, split_points, axis);
980  }

982  template <typename T>
983  std::vector<Tensor<T>> array_split(const Tensor<T>& tensor,
984    size_t n_sections, size_t axis)
985  {
986    auto ndim = tensor.shape().size();
987    if (axis >= ndim)
988      throw std::invalid_argument("array_split: axis " + std::to_string(axis) +
989        " out of range for " + std::to_string(ndim) + "-D tensor");
990    if (n_sections == 0)
991      throw std::invalid_argument("array_split: number of sections must be > 0");
992
993    auto axis_size = tensor.shape()[axis];
994    auto base_size = axis_size / n_sections;
995    auto remainder = axis_size % n_sections;
996
997    std::vector<size_t> split_points;
998    split_points.reserve(n_sections - 1);
999    size_t pos = 0;
1000    for (size_t i = 0; i + 1 < n_sections; ++i)
1001    {
1002      pos += base_size + (i < remainder ? 1 : 0);
1003      split_points.push_back(pos);
1004    }
1005
1006    return split(tensor, split_points, axis);
1007  }

1137  template <typename T>
1138  Tensor<T> Tensor<T>::reshape(const std::vector<ptrdiff_t>& new_shape) const
1139  {
1140    auto total = get_total_size();
1141
1142    // Resolve -1 dimension
1143    std::vector<size_t> resolved(new_shape.size());
1144    ptrdiff_t infer_idx = -1;
1145    size_t known_product = 1;
1146    for (size_t i = 0; i < new_shape.size(); ++i)
1147    {
1148      if (new_shape[i] == -1)
1149      {
1150        if (infer_idx >= 0)
1151          throw std::invalid_argument("Only one dimension can be inferred (-1)");
1152        infer_idx = static_cast<ptrdiff_t>(i);
1153      }
1154      else if (new_shape[i] < 0)
1155      {
1156        throw std::invalid_argument("Invalid dimension size: " + std::to_string(new_shape[i]));
1157      }
1158      else
1159      {
1160        resolved[i] = static_cast<size_t>(new_shape[i]);
1161        known_product *= resolved[i];
1162      }
1163    }
1164
1165    if (infer_idx >= 0)
1166    {
1167      if (known_product == 0 || total % known_product != 0)
1168        throw std::invalid_argument("Cannot infer dimension: total size " +
1169          std::to_string(total) + " is not divisible by " + std::to_string(known_product));
1170      resolved[infer_idx] = total / known_product;
1171    }
1172
1173    // Validate total size
1174    size_t new_total = 1;
1175    for (auto d : resolved) new_total *= d;
1176    if (new_total != total)
1177      throw std::invalid_argument("Cannot reshape tensor of size " +
1178        std::to_string(total) + " into shape " + shape_to_string(resolved));
1179
1180    // Use contiguous source (copy if needed)
1181    Tensor<T> src = m_isContiguous ? *this : copy();
1182
1183    Tensor<T> ret;
1184    ret.m_data = src.m_data;
1185    ret.m_offset = src.m_offset;
1186    ret.m_shape = resolved;
1187    ret.m_isContiguous = true;
1188    ret.m_viewType = ViewType::Base;
1189
1190    // Compute row-major strides
1191    ret.m_strides.resize(resolved.size());
1192    if (!resolved.empty())
1193    {
1194      ret.m_strides.back() = 1;
1195      for (auto i = static_cast<ptrdiff_t>(resolved.size()) - 2; i >= 0; --i)
1196        ret.m_strides[i] = ret.m_strides[i + 1] * static_cast<ptrdiff_t>(resolved[i + 1]);
1197    }
1198
1199    return ret;
1200  }

1277  template <typename T>
1278  Tensor<T> Tensor<T>::transpose(const std::vector<size_t>& axes) const
1279  {
1280    auto ndim = m_shape.size();
1281
1282    std::vector<size_t> perm;
1283    if (axes.empty())
1284    {
1285      perm.resize(ndim);
1286      std::iota(perm.rbegin(), perm.rend(), 0_uz);
1287    }
1288    else
1289    {
1290      if (axes.size() != ndim)
1291        throw std::invalid_argument("transpose: axes must have length " +
1292          std::to_string(ndim) + ", got " + std::to_string(axes.size()));
1293
1294      std::vector<bool> seen(ndim, false);
1295      for (auto a : axes)
1296      {
1297        if (a >= ndim)
1298          throw std::invalid_argument("transpose: axis " + std::to_string(a) + " out of range");
1299        if (seen[a])
1300          throw std::invalid_argument("transpose: repeated axis " + std::to_string(a));
1301        seen[a] = true;
1302      }
1303      perm = axes;
1304    }
1305
1306    Tensor<T> ret;
1307    ret.m_data = m_data;
1308    ret.m_offset = m_offset;
1309    ret.m_shape.resize(ndim);
1310    ret.m_strides.resize(ndim);
1311    for (size_t i = 0; i < ndim; ++i)
1312    {
1313      ret.m_shape[i] = m_shape[perm[i]];
1314      ret.m_strides[i] = m_strides[perm[i]];
1315    }
1316    ret.m_isContiguous = false;
1317    ret.m_viewType = ViewType::Base;
1318
1319    return ret;
1320  }
