40/**
41@class: UnboundedWSQ
42
43@tparam T data type (must be a pointer type)
44
45@brief class to create a lock-free unbounded work-stealing queue
46
47This class implements the work-stealing queue described in the paper,
48<a href="https://www.di.ens.fr/~zappa/readings/ppopp13.pdf">
49Correct and Efficient Work-Stealing for Weak Memory Models</a>.
50
51A work-stealing queue supports a single owner thread that performs push and pop
52operations, while multiple concurrent thief threads may steal tasks
53from the opposite end of the queue. The implementation is designed to
54operate correctly under weak memory models and uses atomic operations
55with carefully chosen memory orderings to ensure correctness and
56scalability.
57
58@dotfile images/unbounded_wsq.dot
59
60Unlike bounded queues, this queue automatically grows its internal
61storage as needed, allowing it to accommodate an arbitrary number of
62tasks without a fixed capacity limit.
63*/
64template <typename T>
65class UnboundedWSQ {
66
67  struct Array {
68
69    size_t C;
70    size_t M;
71    std::atomic<T>* S;
72
73    explicit Array(size_t c) :
74      C {c},
75      M {c-1},
76      S {new std::atomic<T>[C]} {
77    }
78
79    ~Array() {
80      delete [] S;
81    }
82
83    size_t capacity() const noexcept {
84      return C;
85    }
86
87    void push(int64_t i, T o) noexcept {
88      S[i & M].store(o, std::memory_order_relaxed);
89    }
90
91    T pop(int64_t i) noexcept {
92      return S[i & M].load(std::memory_order_relaxed);
93    }
94
95    Array* resize(int64_t b, int64_t t) {
96      Array* ptr = new Array {2*C};
97      for(int64_t i=t; i!=b; ++i) {
98        ptr->push(i, pop(i));
99      }
100      return ptr;
101    }
102
103    Array* resize(int64_t b, int64_t t, size_t N) {
104      // assert(N>0);
105      Array* ptr = new Array {std::bit_ceil(C + N)};
106      for(int64_t i=t; i!=b; ++i) {
107        ptr->push(i, pop(i));
108      }
109      return ptr;
110    }
111
112  };

297  is successful, the stolen task is returned.

301  The elements stolen from the queue follow a first-in-first-out (FIFO) order.
302

318  */
319  value_type steal_with_feedback(size_t& num_empty_steals);
320  
321  /**
322  @brief returns the empty sentinel value for the queue element type
323  
324  This function provides a type-appropriate empty value used to indicate
325  that a pop or steal operation failed. For pointer types, the empty value
326  is `nullptr` of type `T`; for non-pointer types, it is `std::nullopt` of type `std::optional<T>`.
327  
328  The function is implemented as a `constexpr` helper to avoid additional
329  storage, runtime overhead, or code duplication across queue operations.
330

332  */
333  static constexpr auto empty_value() {
334    if constexpr (std::is_pointer_v<T>) {
335      return T{nullptr};
336    } else {
337      return std::optional<T>{std::nullopt};
338    }
339  }

381// Function: push
382template <typename T>
383void UnboundedWSQ<T>::push(T o) {
384
385  int64_t b = _bottom.load(std::memory_order_relaxed);
386  int64_t t = _top.load(std::memory_order_acquire);
387  Array* a = _array.load(std::memory_order_relaxed);
388
389  // queue is full with one additional item (b-t+1)
390  if (a->capacity() < static_cast<size_t>(b - t + 1)) [[unlikely]] {
391    a = _resize_array(a, b, t);
392  }
393
394  a->push(b, o);
395  std::atomic_thread_fence(std::memory_order_release);
396
397  // original paper uses relaxed here but tsa complains
398  _bottom.store(b + 1, std::memory_order_release);
399}

401// Function: bulk_push
402template <typename T>
403template <typename I>
404void UnboundedWSQ<T>::bulk_push(I& first, size_t N) {
405
406  if(N == 0) return;
407
408  int64_t b = _bottom.load(std::memory_order_relaxed);
409  int64_t t = _top.load(std::memory_order_acquire);
410  Array* a = _array.load(std::memory_order_relaxed);
411
412  // queue is full with N additional items
413  if ( (b - t + N) > a->capacity() ) [[unlikely]] {
414    a = _resize_array(a, b, t, N);
415  }
416
417  for(size_t i=0; i<N; ++i) {
418    a->push(b++, *first++);
419  }
420  std::atomic_thread_fence(std::memory_order_release);
421
422  // original paper uses relaxed here but tsa complains
423  _bottom.store(b, std::memory_order_release);
424}

426// Function: pop
427template <typename T>
428typename UnboundedWSQ<T>::value_type 
429UnboundedWSQ<T>::pop() {
430
431  int64_t b = _bottom.load(std::memory_order_relaxed) - 1;
432  Array* a = _array.load(std::memory_order_relaxed);
433  _bottom.store(b, std::memory_order_relaxed);
434  std::atomic_thread_fence(std::memory_order_seq_cst);
435  int64_t t = _top.load(std::memory_order_relaxed);
436
437  //T item {nullptr};
438  auto item = empty_value();
439
440  if(t <= b) {
441    item = a->pop(b);
442    if(t == b) {
443      // the last item just got stolen
444      if(!_top.compare_exchange_strong(t, t+1, std::memory_order_seq_cst,
445                                               std::memory_order_relaxed)) {
446        //item = nullptr;
447        item = empty_value();
448      }
449      _bottom.store(b + 1, std::memory_order_relaxed);
450    }
451  }
452  else {
453    _bottom.store(b + 1, std::memory_order_relaxed);
454  }
455
456  return item;
457}

460template <typename T>

462UnboundedWSQ<T>::steal() {
463  
464  int64_t t = _top.load(std::memory_order_acquire);
465  std::atomic_thread_fence(std::memory_order_seq_cst);
466  int64_t b = _bottom.load(std::memory_order_acquire);
467
468  //T item {nullptr};
469  auto item = empty_value();
470
471  if(t < b) {
472    Array* a = _array.load(std::memory_order_consume);
473    item = a->pop(t);
474    if(!_top.compare_exchange_strong(t, t+1,
475                                     std::memory_order_seq_cst,
476                                     std::memory_order_relaxed)) {
477      //return nullptr;
478      return empty_value();
479    }
480  }
481
482  return item;
483}

746  */

797  }

824  // queue is full with one additional item (b-t+1)

836  return true;

868BoundedWSQ<T, LogSize>::pop() {
869
870  int64_t b = _bottom.load(std::memory_order_relaxed) - 1;
871  _bottom.store(b, std::memory_order_relaxed);
872  std::atomic_thread_fence(std::memory_order_seq_cst);

874

877

880    if(t == b) {

887      }

890  }
