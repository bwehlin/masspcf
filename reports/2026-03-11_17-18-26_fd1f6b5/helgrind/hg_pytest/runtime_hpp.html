11/**
12@class Runtime
13
14@brief class to create a runtime task
15
16A runtime object provides an interface for interacting with the scheduling system from within a task 
17(i.e., the parent task of this runtime). 
18It enables operations such as spawning asynchronous tasks, executing tasks cooperatively, 
19and implementing recursive parallelism. 
20The runtime guarantees an implicit join at the end of its scope, 
21so all spawned tasks will finish before the parent runtime task continues to its successors.
22
23@code{.cpp}
24tf::Executor executor(num_threads);
25tf::Taskflow taskflow;
26std::atomic<size_t> counter(0);
27
28tf::Task A = taskflow.emplace([&](tf::Runtime& rt){
29  // spawn 1000 asynchronous tasks from this runtime task
30  for(size_t i=0; i<1000; i++) {
31    rt.silent_async([&](){ counter.fetch_add(1, std::memory_order_relaxed); });
32  }
33  // implicit synchronization at the end of the runtime scope
34});
35tf::Task B = taskflow.emplace([&](){
36  assert(counter.load(std::memory_order_relaxed) == 1000);
37});
38A.precede(B);
39
40executor.run(taskflow).wait();
41@endcode
42
43@note
44To understand how %Taskflow schedules a runtime task, please refer to @ref RuntimeTasking.
45

638// Procedure: schedule
639inline void Runtime::schedule(Task task) {
640
641  auto node = task._node;
642  // need to keep the invariant: when scheduling a task, the task must have
643  // zero dependency (join counter is 0)
644  // or we can encounter bug when inserting a nested flow (e.g., module task)
645  node->_join_counter.store(0, std::memory_order_relaxed);
646
647  auto& j = node->_parent ? node->_parent->_join_counter :
648                            node->_topology->_join_counter;
649  j.fetch_add(1, std::memory_order_relaxed);
650  _executor._schedule(_worker, node);
651}

653// Function: corun
654inline void Runtime::corun() {
655  {
656    ExplicitAnchorGuard anchor(_node);
657    _executor._corun_until(_worker, [this] () -> bool {
658      return _node->_join_counter.load(std::memory_order_acquire) == 1;
659    });
660  }
661  _node->_rethrow_exception();
662}

852// Function: _invoke_runtime_task_impl
853inline bool Executor::_invoke_runtime_task_impl(
854  Worker& worker, Node* node, std::function<void(Runtime&, bool)>& work
855) {
856    
857  Runtime rt(*this, worker, node);
858
859  // first time
860  if((node->_nstate & NSTATE::PREEMPTED) == 0) {
861    
862    node->_nstate |= (NSTATE::PREEMPTED | NSTATE::IMPLICITLY_ANCHORED);
863    node->_join_counter.fetch_add(1, std::memory_order_release);
864
865    _observer_prologue(worker, node);
866    TF_EXECUTOR_EXCEPTION_HANDLER(worker, node, {
867      work(rt, false);
868    });
869    _observer_epilogue(worker, node);
870    
871    // Last one to leave this runtime; no need to preempt this runtime
872    if(node->_join_counter.fetch_sub(1, std::memory_order_acq_rel) == 1) {
873      node->_nstate &= ~(NSTATE::PREEMPTED | NSTATE::IMPLICITLY_ANCHORED);
874    }
875    // Here, we cannot let caller check the state from node->_nstate due to data race,
876    // but return a stateless boolean to indicate preemption.
877    // Ex: if preempted, another task may finish real quck and insert this parent task
878    // again into the scheduling queue. When running this parent task, it will jump to
879    // else branch below and modify tne nstate, thus incuring data race.
880    else {
881      return true;
882    }
883  }
884  // second time - previously preempted
885  else {
886    node->_nstate &= ~(NSTATE::PREEMPTED | NSTATE::IMPLICITLY_ANCHORED);
887  }
888
889  // clean up outstanding work (e.g., exception)
890  work(rt, true);
891
892  return false;
893}
