31  template <typename Tt, typename Tv, typename RectangleOp>
32  Tv integrate(const Pcf<Tt, Tv>& f, const Pcf<Tt, Tv>& g, RectangleOp op, Tt a = 0.f, Tt b = std::numeric_limits<Tt>::max())
33  {
34    using rect_t = typename Pcf<Tt, Tv>::rectangle_type;
35
36    Tv val = 0.f;
37    iterate_rectangles(f.points(), g.points(), [&val, &op](const rect_t& rect) -> void {
38      val += (rect.right - rect.left) * op(rect);
39    }, a, b);
40    
41    return val;
42  }

45  void make_lower_triangle(Executor& exec, Tv* out, size_t sz)
46  {
