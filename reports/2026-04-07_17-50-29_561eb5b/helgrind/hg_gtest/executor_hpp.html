18/** 
19@class Executor
20
21@brief class to create an executor 
22
23An tf::Executor manages a set of worker threads to run tasks 
24using an efficient work-stealing scheduling algorithm.
25
26@code{.cpp}
27// Declare an executor and a taskflow
28tf::Executor executor;
29tf::Taskflow taskflow;
30
31// Add three tasks into the taskflow
32tf::Task A = taskflow.emplace([] () { std::cout << "This is TaskA\n"; });
33tf::Task B = taskflow.emplace([] () { std::cout << "This is TaskB\n"; });
34tf::Task C = taskflow.emplace([] () { std::cout << "This is TaskC\n"; });
35
36// Build precedence between tasks
37A.precede(B, C);
38
39tf::Future<void> fu = executor.run(taskflow);
40fu.wait();  // block until the execution completes
41
42executor.run(taskflow, [](){ std::cout << "end of 1 run"; }).wait();
43executor.run_n(taskflow, 4);
44executor.wait_for_all();  // block until all associated executions finish
45executor.run_n(taskflow, 4, [](){ std::cout << "end of 4 runs"; }).wait();
46executor.run_until(taskflow, [cnt=0] () mutable { return ++cnt == 10; });
47@endcode
48
49Most executor methods are @em thread-safe. 
50For example, you can submit multiple taskflows to an executor concurrently 
51from different threads, while other threads simultaneously create asynchronous tasks.
52
53@code{.cpp}
54std::thread t1([&](){ executor.run(taskflow); };
55std::thread t2([&](){ executor.async([](){ std::cout << "async task from t2\n"; }); });
56executor.async([&](){ std::cout << "async task from the main thread\n"; });
57@endcode
58
59@note
60To know more about tf::Executor, please refer to @ref ExecuteTaskflow.
61*/

414  @brief runs a target graph and waits until it completes using 
415         an internal worker of this executor
416  
417  @tparam T target type which has `tf::Graph& T::graph()` defined
418  @param target the target task graph object
419
420  The method coruns a target graph cooperatively with other workers in the same executor
421  and block until the execution completes.

1841// Procedure: _process_exception
1842inline void Executor::_process_exception(Worker&, Node* node) {
1843
1844  // Finds the anchor and mark the entire path with exception, 
1845  // so recursive tasks can be cancelled properly.
1846  // Since exception can come from asynchronous task (with runtime), the node itself can be anchored.
1847  NodeBase* ea = node;     // explicit anchor
1848  NodeBase* ia = nullptr;  // implicit anchor
1849  
1850  while(ea && (ea->_estate.load(std::memory_order_relaxed) & ESTATE::EXPLICITLY_ANCHORED) == 0) {
1851    ea->_estate.fetch_or(ESTATE::EXCEPTION, std::memory_order_relaxed);
1852    // we only want the inner-most implicit anchor
1853    if(ia == nullptr && (ea->_nstate & NSTATE::IMPLICITLY_ANCHORED)) {
1854      ia = ea;
1855    }
1856    ea = ea->_parent;
1857  }
1858  
1859  // flag used to ensure execution is caught in a thread-safe manner
1860  constexpr static auto flag = ESTATE::EXCEPTION | ESTATE::CAUGHT;
1861
1862  // The exception occurs under a blocking call (e.g., corun, join).
1863  if(ea) {
1864    // multiple tasks may throw, and we only take the first thrown exception
1865    if((ea->_estate.fetch_or(flag, std::memory_order_relaxed) & ESTATE::CAUGHT) == 0) {
1866      ea->_exception_ptr = std::current_exception();
1867      return;
1868    }
1869  }
1870  // Implicit anchor has the lowest priority
1871  else if(ia){
1872    if((ia->_estate.fetch_or(flag, std::memory_order_relaxed) & ESTATE::CAUGHT) == 0) {
1873      ia->_exception_ptr = std::current_exception();
1874      return;
1875    }
1876  }
1877  
1878  // For now, we simply store the exception in this node; this can happen in an 
1879  // execution that does not have any external control to capture the exception,
1880  // such as silent async task without any parent.
1881  node->_exception_ptr = std::current_exception();
1882}
