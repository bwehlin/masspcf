96    B2_t beg2 = first2;
97    O_t d_beg = d_first;
98
99    size_t W = rt.executor().num_workers();
100    size_t N = std::distance(beg1, end1);
101
102    // only myself - no need to spawn another graph
103    if(W <= 1 || N <= part.chunk_size()) {
104      part([=]() mutable { std::transform(beg1, end1, beg2, d_beg, c); })();
105      return;
106    }
107    
108    if(N < W) {
109      W = N;
110    }
111
112    // static partitioner
113    if constexpr(part.type() == PartitionerType::STATIC) {
114      for(size_t w=0, curr_b=0; w<W && curr_b < N;) {
115        auto chunk_size = part.adjusted_chunk_size(N, W, w);
116        auto task = part([=] () mutable {
117          part.loop(N, W, curr_b, chunk_size, [=, prev_e=size_t{0}](size_t part_b, size_t part_e) mutable {
118            std::advance(beg1, part_b - prev_e);
119            std::advance(beg2, part_b - prev_e);
120            std::advance(d_beg, part_b - prev_e);
121            for(size_t x = part_b; x<part_e; x++) {
122              *d_beg++ = c(*beg1++, *beg2++);
123            }
124            prev_e = part_e;
125          });
126        });
127        (++w == W || (curr_b += chunk_size) >= N) ? task() : rt.silent_async(task);
128      }
129    }
130    // dynamic partitioner
131    else {
132      auto next = std::make_shared<std::atomic<size_t>>(0);
133      for(size_t w=0; w<W;) {
134        auto task = part([=] () mutable {
135          part.loop(N, W, *next, [=, prev_e=size_t{0}](size_t part_b, size_t part_e) mutable {
136            std::advance(beg1, part_b - prev_e);
137            std::advance(beg2, part_b - prev_e);
138            std::advance(d_beg, part_b - prev_e);
139            for(size_t x = part_b; x<part_e; x++) {
140              *d_beg++ = c(*beg1++, *beg2++);
141            }
142            prev_e = part_e;
143          });
144        });
145        (++w == W) ? task() : rt.silent_async(task);
146      }
147    }
148  };
149}
