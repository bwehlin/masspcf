39  template <typename TPcf>
40  TPcf combine(const TPcf& f, const TPcf& g, TOp<TPcf> op)
41  {
42    using point_type = typename TPcf::point_type;
43    using rectangle_type = typename TPcf::rectangle_type;
44    using value_type = typename point_type::value_type;
45
46    std::vector<point_type> retPts;
47
48    std::size_t npts = 0_uz;
49    iterate_rectangles(f.points(), g.points(), [&npts](const rectangle_type&){ ++npts; });
50    retPts.reserve(npts);
51
52    value_type vLast = static_cast<value_type>(0);
53    bool first = true;
54
55    iterate_rectangles(f.points(), g.points(), [&retPts, &op, &vLast, &first](const rectangle_type& rect){
56      auto v = op(rect);
57
58      // Avoid inserting duplicate plateaus with the same value, except at t=0
59      if (first || v != vLast || rect.left == 0)
60      {
61        retPts.emplace_back(rect.left, v);
62        first = false;
63      }
64
65      vLast = v;
66    });
67
68    retPts.shrink_to_fit();
69
70    return TPcf(std::move(retPts));
71  }
