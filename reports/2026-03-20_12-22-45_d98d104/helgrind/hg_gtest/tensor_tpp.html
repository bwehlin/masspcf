23    : m_shape(shape)
24  {
25    auto sz = get_total_size();

27    std::fill_n(m_data.get(), sz, init);
28
29    // Compute strides
30    if (!m_shape.empty())
31    {
32      m_strides = m_shape;
33
34      std::partial_sum(m_shape.rbegin(), std::prev(m_shape.rend()), std::next(m_strides.rbegin()), std::multiplies<>());
35      m_strides.back() = 1;
36    }
37  }

313  template <typename T>
314  Tensor<T> Tensor<T>::broadcast_to(const std::vector<size_t>& target_shape) const
315  {
316    size_t ndim = target_shape.size();
317    size_t src_ndim = m_shape.size();
318
319    if (ndim < src_ndim)
320    {
321      throw std::invalid_argument("Cannot broadcast shape " +
322          shape_to_string(m_shape) + " to " + shape_to_string(target_shape) +
323          ": target has fewer dimensions");
324    }
325
326    Tensor ret;
327    ret.m_data = m_data;
328    ret.m_offset = m_offset;
329    ret.m_shape = target_shape;
330    ret.m_strides.resize(ndim);
331    ret.m_isContiguous = false;
332
333    size_t prepended = ndim - src_ndim;
334    for (size_t i = 0; i < ndim; ++i)
335    {
336      if (i < prepended)
337      {
338        ret.m_strides[i] = 0;
339      }
340      else
341      {
342        size_t si = i - prepended;
343        if (m_shape[si] == target_shape[i])
344        {
345          ret.m_strides[i] = m_strides[si];
346        }
347        else if (m_shape[si] == 1)
348        {
349          ret.m_strides[i] = 0;
350        }
351        else
352        {
353          throw std::invalid_argument("Cannot broadcast shape " +
354              shape_to_string(m_shape) + " to " + shape_to_string(target_shape));
355        }
356      }
357    }
358
359    return ret;
360  }

366    /// Apply a binary operation elementwise to two broadcast-compatible tensors.
367    /// The result element type R is deduced from the return type of op.
368    template <typename T, typename BinaryOp,
369              typename R = std::invoke_result_t<BinaryOp, const T&, const T&>>
370    Tensor<R> broadcast_binop(const Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
371    {
372      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
373      auto lhs_view = lhs.broadcast_to(out_shape);
374      auto rhs_view = rhs.broadcast_to(out_shape);
375      Tensor<R> result(out_shape);
376      result.walk([&](const std::vector<size_t>& idx) {
377        result(idx) = op(lhs_view(idx), rhs_view(idx));
378      });
379      return result;
380    }
381
382    template <typename T, typename BinaryOp>
383    Tensor<T>& broadcast_binop_inplace(Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
384    {
385      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
386      if (out_shape != lhs.shape())
387        throw std::invalid_argument("Cannot broadcast in-place: output shape " +
388            shape_to_string(out_shape) + " differs from LHS shape " + shape_to_string(lhs.shape()));
389      auto rhs_view = rhs.broadcast_to(out_shape);
390      lhs.walk([&](const std::vector<size_t>& idx) {
391        lhs(idx) = op(lhs(idx), rhs_view(idx));
392      });
393      return lhs;
394    }
395  }

556  template <typename T>
557  template <typename SliceVector>
558  Tensor<T> Tensor<T>::extract(SliceVector sliceVector) const
559  {
560    Tensor ret;
561
562    ret.m_data = m_data; // view
563
564    // temporary just to get sizes
565    ret.m_shape = m_shape;
566    ret.m_strides = m_strides;
567
568    ret.m_offset = m_offset;
569
570    std::vector<size_t> dimsToDrop;
571
572    // For now, we'll drop the assumption that the tensor is contiguous in memory
573    // as soon as we extract a subtensor. There are, however, some cases where the resulting subtensor would be
574    // contiguous that we can try to optimize for in the future (e.g., extracting the top n rows of a matrix).
575    // Things will work fine with this assumption dropped but certain operations could be a little slower (probably
576    // unlikely to matter for the type of things we're targeting).
577    ret.m_isContiguous = false;
578
579    size_t i = 0;
580    for (auto & slice : sliceVector)
581    {
582      std::visit([i, &slice, &ret, &dimsToDrop](auto&& arg) {
583        using argT = std::decay_t<decltype(arg)>;
584        if constexpr (std::is_same_v<argT, SliceIndex>)
585        {
586          ret.m_shape[i] = 1;
587          ret.m_offset += arg.index * ret.m_strides[i];
588          dimsToDrop.emplace_back(i);
589        }
590        else if constexpr (std::is_same_v<argT, SliceRange>)
591        {
592          if (!arg.start)
593          {
594            arg.start = 0_z;
595          }
596          if (!arg.stop)
597          {
598            arg.stop = static_cast<ptrdiff_t>(ret.m_shape[i]);
599          }
600          if (!arg.step)
601          {
602            arg.step = 1_z;
603          }
604
605          auto start = *arg.start;
606          auto stop = *arg.stop;
607          auto step = *arg.step;
608
609          if (start < 0_z)
610            start = 0;
611          if (stop > static_cast<std::ptrdiff_t>(ret.m_shape[i]))
612            stop = static_cast<std::ptrdiff_t>(ret.m_shape[i]);
613
614          if (step == 0_z)
615          {
616            ret.m_shape[i] = 0;
617          }
618          else if (step > 0)
619          {
620            if (stop <= start)
621            {
622              ret.m_shape[i] = 0;
623            }
624            else
625            {
626              ret.m_shape[i] = (stop - start + step - 1_z) / step;
627            }
628          }
629          else
630          {
631            throw std::runtime_error("Negative step not supported...");
632          }
633
634          ret.m_offset += start * ret.m_strides[i]; // use clamped start
635          ret.m_strides[i] *= step;
636
637        }
638        // For SliceAll, don't modify shape
639      }, slice);
640      ++i;
641    }
642
643    size_t nDroppedDims = 0_uz;
644    for (auto dim : dimsToDrop)
645    {
646      ret.m_shape.erase(ret.m_shape.begin() + dim - nDroppedDims);
647      ret.m_strides.erase(ret.m_strides.begin() + dim - nDroppedDims);
648      ++nDroppedDims;
649    }
650
651    return ret;
652  }

709
