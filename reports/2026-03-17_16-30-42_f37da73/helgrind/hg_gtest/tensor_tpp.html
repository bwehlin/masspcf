260  template <typename T>
261  Tensor<T> Tensor<T>::broadcast_to(const std::vector<size_t>& target_shape) const
262  {
263    size_t ndim = target_shape.size();
264    size_t src_ndim = m_shape.size();
265
266    if (ndim < src_ndim)
267    {
268      throw std::invalid_argument("Cannot broadcast shape " +
269          shape_to_string(m_shape) + " to " + shape_to_string(target_shape) +
270          ": target has fewer dimensions");
271    }
272
273    Tensor ret;
274    ret.m_data = m_data;
275    ret.m_offset = m_offset;
276    ret.m_shape = target_shape;
277    ret.m_strides.resize(ndim);
278    ret.m_isContiguous = false;
279
280    size_t prepended = ndim - src_ndim;
281    for (size_t i = 0; i < ndim; ++i)
282    {
283      if (i < prepended)
284      {
285        ret.m_strides[i] = 0;
286      }
287      else
288      {
289        size_t si = i - prepended;
290        if (m_shape[si] == target_shape[i])
291        {
292          ret.m_strides[i] = m_strides[si];
293        }
294        else if (m_shape[si] == 1)
295        {
296          ret.m_strides[i] = 0;
297        }
298        else
299        {
300          throw std::invalid_argument("Cannot broadcast shape " +
301              shape_to_string(m_shape) + " to " + shape_to_string(target_shape));
302        }
303      }
304    }
305
306    return ret;
307  }

313    template <typename T, typename BinaryOp>
314    Tensor<T> broadcast_binop(const Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
315    {
316      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
317      auto lhs_view = lhs.broadcast_to(out_shape);
318      auto rhs_view = rhs.broadcast_to(out_shape);
319      Tensor<T> result(out_shape);
320      result.walk([&](const std::vector<size_t>& idx) {
321        result(idx) = op(lhs_view(idx), rhs_view(idx));
322      });
323      return result;
324    }
325
326    template <typename T, typename BinaryOp>
327    Tensor<T>& broadcast_binop_inplace(Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
328    {
329      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
330      if (out_shape != lhs.shape())
331        throw std::invalid_argument("Cannot broadcast in-place: output shape " +
332            shape_to_string(out_shape) + " differs from LHS shape " + shape_to_string(lhs.shape()));
333      auto rhs_view = rhs.broadcast_to(out_shape);
334      lhs.walk([&](const std::vector<size_t>& idx) {
335        lhs(idx) = op(lhs(idx), rhs_view(idx));
336      });
337      return lhs;
338    }

414  template <typename T>
415  template <typename UnaryFunc>
416#ifndef __CUDACC__
417  requires std::invocable<UnaryFunc, std::vector<size_t>>
418#endif
419  void Tensor<T>::walk(UnaryFunc&& f) const
420  {
421    if (m_shape.empty() || std::any_of(m_shape.begin(), m_shape.end(), [](size_t n){ return n == 0; }))
422    {
423      return;
424    }
425    auto ndim = m_shape.size();
426
427    std::vector<size_t> cur(ndim, 0_uz);
428
429    while (true)
430    {
431      if constexpr (std::is_same_v<decltype(f(cur)), bool>)
432      {
433        // If f returns bool, stop walking on false being returned.
434        if (!f(cur))
435        {
436          return;
437        }
438      }
439      else
440      {
441        f(cur);
442      }
443
444
445      for (ptrdiff_t i = ndim - 1; i >= 0 ; --i)
446      {
447        ++cur[i];
448
449        if (cur[i] < m_shape[i])
450        {
451          break;
452        }
453
454        if (i == 0)
455        {
456          return;
457        }
458
459        cur[i] = 0;
460      }
461    }
462  }

508  template <typename T>
509  template <typename SliceVector>
510  Tensor<T> Tensor<T>::extract(SliceVector sliceVector) const
511  {
512    Tensor ret;
513
514    ret.m_data = m_data; // view
515
516    // temporary just to get sizes
517    ret.m_shape = m_shape;
518    ret.m_strides = m_strides;
519
520    ret.m_offset = m_offset;
521
522    std::vector<size_t> dimsToDrop;
523
524    // For now, we'll drop the assumption that the tensor is contiguous in memory
525    // as soon as we extract a subtensor. There are, however, some cases where the resulting subtensor would be
526    // contiguous that we can try to optimize for in the future (e.g., extracting the top n rows of a matrix).
527    // Things will work fine with this assumption dropped but certain operations could be a little slower (probably
528    // unlikely to matter for the type of things we're targeting).
529    ret.m_isContiguous = false;
530
531    size_t i = 0;
532    for (auto & slice : sliceVector)
533    {
534      std::visit([i, &slice, &ret, &dimsToDrop](auto&& arg) {
535        using argT = std::decay_t<decltype(arg)>;
536        if constexpr (std::is_same_v<argT, SliceIndex>)
537        {
538          ret.m_shape[i] = 1;
539          ret.m_offset += arg.index * ret.m_strides[i];
540          dimsToDrop.emplace_back(i);
541        }
542        else if constexpr (std::is_same_v<argT, SliceRange>)
543        {
544          if (!arg.start)
545          {
546            arg.start = 0_z;
547          }
548          if (!arg.stop)
549          {
550            arg.stop = static_cast<ptrdiff_t>(ret.m_shape[i]);
551          }
552          if (!arg.step)
553          {
554            arg.step = 1_z;
555          }
556
557          auto start = *arg.start;
558          auto stop = *arg.stop;
559          auto step = *arg.step;
560
561          if (start < 0_z)
562            start = 0;
563          if (stop > static_cast<std::ptrdiff_t>(ret.m_shape[i]))
564            stop = static_cast<std::ptrdiff_t>(ret.m_shape[i]);
565
566          if (step == 0_z)
567          {
568            ret.m_shape[i] = 0;
569          }
570          else if (step > 0)
571          {
572            if (stop <= start)
573            {
574              ret.m_shape[i] = 0;
575            }
576            else
577            {
578              ret.m_shape[i] = (stop - start + step - 1_z) / step;
579            }
580          }
581          else
582          {
583            throw std::runtime_error("Negative step not supported...");
584          }
585
586          ret.m_offset += start * ret.m_strides[i]; // use clamped start
587          ret.m_strides[i] *= step;
588
589        }
590        // For SliceAll, don't modify shape
591      }, slice);
592      ++i;
593    }
594
595    size_t nDroppedDims = 0_uz;
596    for (auto dim : dimsToDrop)
597    {
598      ret.m_shape.erase(ret.m_shape.begin() + dim - nDroppedDims);
599      ret.m_strides.erase(ret.m_strides.begin() + dim - nDroppedDims);
600      ++nDroppedDims;
601    }
602
603    return ret;
604  }
