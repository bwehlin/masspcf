268  }

270  template <typename Tt, typename Tv>
271  void PrintTo(const Pcf<Tt, Tv>& f, std::ostream* os)
272  {
273    *os << f.to_string();
274  }

276  template <typename F>
277  concept PcfLike = requires(F f, const F cf)
278  {
279    typename F::point_type;
280    typename F::time_type;
281    typename F::value_type;
282    typename F::rectangle_type;
283    typename F::segment_type;
284
285    requires PointLike<typename F::point_type>;
286
287    { cf.points() } -> std::convertible_to<const std::vector<typename F::point_type>&>;
288    { cf.size()   } -> std::convertible_to<std::size_t>;
289
290    { cf == cf } -> std::convertible_to<bool>;
291    { cf != cf } -> std::convertible_to<bool>;
292  };
