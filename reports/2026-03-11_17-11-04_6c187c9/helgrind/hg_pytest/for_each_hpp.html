146auto make_for_each_by_index_task(R range, C c, P part = P()){
147  
148  using range_type = std::decay_t<unwrap_ref_decay_t<R>>;
149
150  return [=] (Runtime& rt) mutable {
151
152    // fetch the iterator values
153    range_type r = range;
154    
155    // nothing to be done if the range is invalid
156    if(is_index_range_invalid(r.begin(), r.end(), r.step_size())) {
157      return;
158    }
159
160    size_t W = rt.executor().num_workers();
161    size_t N = r.size();
162
163    // only myself - no need to spawn another graph
164    if(W <= 1 || N <= part.chunk_size()) {
165      part([=]() mutable { c(r); })();
166      return;
167    }
168
169    if(N < W) {
170      W = N;
171    }
172    
173    // static partitioner
174    if constexpr(part.type() == PartitionerType::STATIC) {
175      for(size_t w=0, curr_b=0; w<W && curr_b < N;) {
176        auto chunk_size = part.adjusted_chunk_size(N, W, w);
177        auto task = part([=] () mutable {
178          part.loop(N, W, curr_b, chunk_size, [=] (size_t part_b, size_t part_e) {
179            c(r.discrete_domain(part_b, part_e));
180          });
181        });
182        (++w == W || (curr_b += chunk_size) >= N) ? task() : rt.silent_async(task);
183      }
184    }
185    // dynamic partitioner
186    else {

188      for(size_t w=0; w<W;) {
189        auto task = part([=] () mutable {
190          part.loop(N, W, *next, [=] (size_t part_b, size_t part_e) {
191            c(r.discrete_domain(part_b, part_e));
192          });
193        });
194        (++w == W) ? task() : rt.silent_async(task);
195      }
196    }
