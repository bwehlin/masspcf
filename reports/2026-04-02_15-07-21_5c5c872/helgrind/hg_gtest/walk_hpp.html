32    /**
33     * Core sequential walk: visits every index in row-major order, providing
34     * both the multi-index and the flat (row-major) counter to the callback.
35     * If f returns bool, walking stops when f returns false.
36     */
37    template <IsTensor TTensor, typename Func>
38    void walk_impl(const TTensor& tensor, Func&& f)
39    {
40      auto shape_range = tensor.shape();
41      std::vector<size_t> shape(std::begin(shape_range), std::end(shape_range));
42
43      if (shape.empty() || std::any_of(shape.begin(), shape.end(), [](size_t n){ return n == 0; }))
44      {
45        return;
46      }
47
48      auto ndim = shape.size();
49      std::vector<size_t> cur(ndim, 0_uz);
50      size_t flat = 0;
51
52      while (true)
53      {
54        if constexpr (std::is_same_v<decltype(f(cur, flat)), bool>)
55        {
56          if (!f(cur, flat))
57          {
58            return;
59          }
60        }
61        else
62        {
63          f(cur, flat);
64        }
65
66        ++flat;
67
68        for (ptrdiff_t i = ndim - 1; i >= 0; --i)
69        {
70          ++cur[i];
71
72          if (cur[i] < shape[i])
73          {
74            break;
75          }
76
77          if (i == 0)
78          {
79            return;
80          }
81
82          cur[i] = 0;
83        }
84      }
85    }

133  /**
134   * Visit every index of any IsTensor in row-major order, invoking f(idx) at each.
135   * If f returns bool, walking stops when f returns false.
136   */
137  template <IsTensor TTensor, typename UnaryFunc>

139  requires std::invocable<UnaryFunc, std::vector<size_t>>
140#endif
141  void walk(const TTensor& tensor, UnaryFunc&& f)
142  {
143    detail::walk_impl(tensor, [&f](const std::vector<size_t>& idx, size_t) -> decltype(auto) {
144      return f(idx);

146  }

179  /**
180   * Walk every index of a tensor, providing a deterministically-seeded random
181   * engine at each element. The lambda receives (idx, engine&) where the engine
182   * is seeded from the generator and the element's flat (row-major) index.
183   */
184  template <IsTensor TTensor, typename EngineT, typename BinaryFunc>
185#ifndef __CUDACC__
186  requires std::invocable<BinaryFunc, std::vector<size_t>, EngineT&>
187#endif
188  void walk(const TTensor& tensor, const RandomGenerator<EngineT>& gen, BinaryFunc&& f)
189  {
190    detail::walk_impl(tensor, [&gen, &f](const std::vector<size_t>& idx, size_t flat) {
191      auto engine = gen.sub_generator(flat);
192      f(idx, engine);
193    });
