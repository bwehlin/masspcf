200    [[nodiscard]] value_type evaluate(time_type t) const
201    {
202      auto it = std::upper_bound(m_points.begin(), m_points.end(), t,
203        [](time_type time, const point_type& pt) { return time < pt.t; });
204
205      if (it == m_points.begin())
206      {
207        throw std::domain_error("Cannot evaluate PCF before time 0.");
208      }
209
210      --it;
211      return it->v;
212    }

214    template <typename TIn, typename TOut>
215    void evaluate(const TIn& sorted_times, TOut& out, size_t n) const
216    {
217      auto it = m_points.begin();
218      auto end = m_points.end();
219
220      for (size_t i = 0; i < n; ++i)
221      {
222        auto t = sorted_times(i);
223
224        if (t < m_points.front().t)
225        {
226          throw std::domain_error("Cannot evaluate PCF before time 0.");
227        }
228
229        while (std::next(it) != end && std::next(it)->t <= t)
230        {
231          ++it;
232        }
233
234        out(i) = it->v;
235      }
236    }

238    /**
239     * Raise every value of the PCF to a power.
240     *
241     * Returns a new PCF whose value at every breakpoint is raised to
242     * `exponent`. The domain (time coordinates) is unchanged.
243     * Squaring (exponent == 2) avoids `std::pow` for performance.
244     *
245     * @param exponent the exponent to raise each value to
246     * @return a new PCF with transformed values
247     */
248    template <Arithmetic T>
249    [[nodiscard]] Pcf pow(T exponent) const
250    {
251      auto pts = m_points;
252      if (exponent == 2)
253      {
254        for (auto& pt : pts)
255        {
256          pt.v = pt.v * pt.v;
257        }
258      }
259      else
260      {
261        for (auto& pt : pts)
262        {
263          pt.v = static_cast<Tv>(std::pow(pt.v, exponent));
264        }
265      }
266      return Pcf(std::move(pts));
267    }
