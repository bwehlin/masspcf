43    : m_shape(shape)
44  {

55        m_strides[i] = m_strides[i + 1] * static_cast<ptrdiff_t>(m_shape[i + 1]);

351  template <typename T>
352  Tensor<T> Tensor<T>::broadcast_to(const std::vector<size_t>& target_shape) const
353  {
354    size_t ndim = target_shape.size();
355    size_t src_ndim = m_shape.size();
356
357    if (ndim < src_ndim)
358    {
359      throw std::invalid_argument("Cannot broadcast shape " +
360          shape_to_string(m_shape) + " to " + shape_to_string(target_shape) +
361          ": target has fewer dimensions");
362    }
363
364    Tensor ret;
365    ret.m_data = m_data;
366    ret.m_offset = m_offset;
367    ret.m_shape = target_shape;
368    ret.m_strides.resize(ndim);
369    ret.m_isContiguous = false;
370
371    size_t prepended = ndim - src_ndim;
372    for (size_t i = 0; i < ndim; ++i)
373    {
374      if (i < prepended)
375      {
376        ret.m_strides[i] = 0;
377      }
378      else
379      {
380        size_t si = i - prepended;
381        if (m_shape[si] == target_shape[i])
382        {
383          ret.m_strides[i] = m_strides[si];
384        }
385        else if (m_shape[si] == 1)
386        {
387          ret.m_strides[i] = 0;
388        }
389        else
390        {
391          throw std::invalid_argument("Cannot broadcast shape " +
392              shape_to_string(m_shape) + " to " + shape_to_string(target_shape));
393        }
394      }
395    }
396
397    return ret;
398  }

404    /// Apply a binary operation elementwise to two broadcast-compatible tensors.
405    /// The result element type R is deduced from the return type of op.
406    template <typename T, typename BinaryOp,
407              typename R = std::invoke_result_t<BinaryOp, const T&, const T&>>
408    Tensor<R> broadcast_binop(const Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
409    {
410      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
411      auto lhs_view = lhs.broadcast_to(out_shape);
412      auto rhs_view = rhs.broadcast_to(out_shape);
413      Tensor<R> result(out_shape);
414      mpcf::walk(result, [&](const std::vector<size_t>& idx) {
415        result(idx) = op(lhs_view(idx), rhs_view(idx));
416      });
417      return result;
418    }
419
420    template <typename T, typename BinaryOp>
421    Tensor<T>& broadcast_binop_inplace(Tensor<T>& lhs, const Tensor<T>& rhs, BinaryOp op)
422    {
423      auto out_shape = broadcast_shapes(lhs.shape(), rhs.shape());
424      if (out_shape != lhs.shape())
425        throw std::invalid_argument("Cannot broadcast in-place: output shape " +
426            shape_to_string(out_shape) + " differs from LHS shape " + shape_to_string(lhs.shape()));
427      auto rhs_view = rhs.broadcast_to(out_shape);
428      mpcf::walk(lhs, [&](const std::vector<size_t>& idx) {
429        lhs(idx) = op(lhs(idx), rhs_view(idx));
430      });
431      return lhs;
432    }

508  template <typename T>
509  Tensor<T> masked_select(const Tensor<T>& src, const Tensor<bool>& mask)
510  {
511    if (mask.shape() != src.shape())
512    {
513      throw std::invalid_argument(
514        "masked_select: mask shape does not match tensor shape");
515    }
516
517    // Single pass: collect matching elements into a buffer
518    std::vector<T> buf;
519    walk(src, [&](const std::vector<size_t>& idx) {
520      if (mask(idx))
521        buf.push_back(src(idx));
522    });
523
524    if (buf.empty())
525    {
526      return Tensor<T>({0});
527    }
528
529    Tensor<T> result({buf.size()});
530    std::move(buf.begin(), buf.end(), result.data());
531
532    return result;
533  }

609  template <typename T>
610  Tensor<T> axis_select(const Tensor<T>& src, size_t axis, const Tensor<bool>& mask)
611  {
612    detail::validate_axis_mask(axis, src.shape().size(), mask, src.shape()[axis]);
613    auto true_indices = detail::collect_true_indices(mask);
614
615    // Build output shape
616    auto out_shape = src.shape();
617    out_shape[axis] = true_indices.size();
618
619    if (true_indices.empty())
620    {
621      return Tensor<T>(out_shape);
622    }
623
624    Tensor<T> result(out_shape);
625    walk(result, [&](const std::vector<size_t>& out_idx) {
626      auto src_idx = out_idx;
627      src_idx[axis] = true_indices[out_idx[axis]];
628      result(out_idx) = src(src_idx);
629    });
630
631    return result;
632  }

751    struct ResolvedSelector
752    {
753      size_t axis;
754      std::vector<size_t> indices;  // selected positions along this axis
755    };
756
757    inline std::vector<ResolvedSelector> resolve_selectors(
758      const std::vector<std::pair<size_t, AxisSelector>>& selectors,
759      const std::vector<size_t>& shape)
760    {
761      std::vector<ResolvedSelector> result;
762      result.reserve(selectors.size());
763      for (const auto& [axis, sel] : selectors)
764      {
765        if (axis >= shape.size())
766          throw std::invalid_argument("outer indexing: axis " + std::to_string(axis) +
767            " out of range for tensor with " + std::to_string(shape.size()) + " dimensions");
768
769        ResolvedSelector rs;
770        rs.axis = axis;
771        std::visit([&](const auto& idx_tensor) {
772          using IdxT = std::decay_t<decltype(idx_tensor)>;
773          if (idx_tensor.shape().size() != 1)
774            throw std::invalid_argument("outer indexing: selector must be 1D");
775
776          if constexpr (std::is_same_v<IdxT, Tensor<bool>>)
777          {
778            if (idx_tensor.shape()[0] != shape[axis])
779              throw std::invalid_argument("outer indexing: bool mask length must match axis size");
780            rs.indices = collect_true_indices(idx_tensor);
781          }
782          else
783          {
784            rs.indices.reserve(idx_tensor.shape()[0]);
785            for (size_t i = 0; i < idx_tensor.shape()[0]; ++i)
786              rs.indices.push_back(static_cast<size_t>(idx_tensor({i})));
787          }
788        }, sel);
789        result.push_back(std::move(rs));
790      }
791      return result;
792    }
793  }

887  template <typename T>
888  Tensor<T> concatenate(const std::vector<Tensor<T>>& tensors, size_t axis)
889  {
890    if (tensors.empty())
891      throw std::invalid_argument("concatenate: need at least one tensor");
892
893    auto ndim = tensors[0].shape().size();
894    if (axis >= ndim)
895      throw std::invalid_argument("concatenate: axis " + std::to_string(axis) +
896        " out of range for " + std::to_string(ndim) + "-D tensor");
897
898    // Validate shapes match on all axes except the join axis
899    for (size_t t = 1; t < tensors.size(); ++t)
900    {
901      if (tensors[t].shape().size() != ndim)
902        throw std::invalid_argument("concatenate: all tensors must have the same number of dimensions");
903      for (size_t d = 0; d < ndim; ++d)
904      {
905        if (d != axis && tensors[t].shape()[d] != tensors[0].shape()[d])
906          throw std::invalid_argument("concatenate: shape mismatch on dimension " + std::to_string(d));
907      }
908    }
909
910    // Compute output shape
911    auto out_shape = tensors[0].shape();
912    out_shape[axis] = 0;
913    for (const auto& t : tensors)
914      out_shape[axis] += t.shape()[axis];
915
916    Tensor<T> result(out_shape);
917
918    // Copy each tensor's data into the result
919    size_t offset = 0;
920    for (const auto& src : tensors)
921    {
922      auto src_axis_size = src.shape()[axis];
923      walk(src, [&](const std::vector<size_t>& src_idx) {
924        auto dst_idx = src_idx;
925        dst_idx[axis] += offset;
926        result(dst_idx) = src(src_idx);
927      });
928      offset += src_axis_size;
929    }
930
931    return result;
932  }

934  template <typename T>
935  Tensor<T> stack(const std::vector<Tensor<T>>& tensors, ptrdiff_t axis)
936  {
937    if (tensors.empty())
938      throw std::invalid_argument("stack: need at least one tensor");
939
940    auto ndim = static_cast<ptrdiff_t>(tensors[0].shape().size());
941
942    // Resolve negative axis; valid range for new axis is [-(ndim+1), ndim]
943    if (axis < 0)
944      axis += ndim + 1;
945    if (axis < 0 || axis > ndim)
946      throw std::invalid_argument("stack: axis out of range");
947
948    // Validate all shapes match
949    for (size_t t = 1; t < tensors.size(); ++t)
950    {
951      if (tensors[t].shape() != tensors[0].shape())
952        throw std::invalid_argument("stack: all tensors must have the same shape");
953    }
954
955    // expand_dims each tensor, then concatenate along the new axis
956    std::vector<Tensor<T>> expanded;
957    expanded.reserve(tensors.size());
958    for (const auto& t : tensors)
959      expanded.push_back(t.expand_dims(axis));
960
961    return concatenate(expanded, static_cast<size_t>(axis));
962  }

964  template <typename T>
965  std::vector<Tensor<T>> split(const Tensor<T>& tensor,
966    const std::vector<size_t>& split_points, size_t axis)
967  {
968    auto ndim = tensor.shape().size();
969    if (axis >= ndim)
970      throw std::invalid_argument("split: axis " + std::to_string(axis) +
971        " out of range for " + std::to_string(ndim) + "-D tensor");
972
973    auto axis_size = tensor.shape()[axis];
974
975    // Build slice boundaries: [0, sp0, sp1, ..., axis_size]
976    std::vector<size_t> boundaries;
977    boundaries.reserve(split_points.size() + 2);
978    boundaries.push_back(0);
979    for (auto sp : split_points)
980    {
981      if (sp > axis_size)
982        sp = axis_size;
983      boundaries.push_back(sp);
984    }
985    boundaries.push_back(axis_size);
986
987    std::vector<Tensor<T>> result;
988    result.reserve(boundaries.size() - 1);
989    for (size_t i = 0; i + 1 < boundaries.size(); ++i)
990    {
991      std::vector<Slice> slices(ndim, all());
992      slices[axis] = range(
993        static_cast<ptrdiff_t>(boundaries[i]),
994        static_cast<ptrdiff_t>(boundaries[i + 1]),
995        std::nullopt);
996      result.push_back(tensor[slices]);
997    }
998    return result;
999  }

1001  template <typename T>
1002  std::vector<Tensor<T>> split(const Tensor<T>& tensor,
1003    size_t n_sections, size_t axis)
1004  {
1005    auto ndim = tensor.shape().size();
1006    if (axis >= ndim)
1007      throw std::invalid_argument("split: axis " + std::to_string(axis) +
1008        " out of range for " + std::to_string(ndim) + "-D tensor");
1009
1010    auto axis_size = tensor.shape()[axis];
1011    if (axis_size % n_sections != 0)
1012      throw std::invalid_argument("split: tensor of size " +
1013        std::to_string(axis_size) + " along axis " + std::to_string(axis) +
1014        " cannot be evenly split into " + std::to_string(n_sections) + " sections");
1015
1016    auto section_size = axis_size / n_sections;
1017    std::vector<size_t> split_points;
1018    split_points.reserve(n_sections - 1);
1019    for (size_t i = 1; i < n_sections; ++i)
1020      split_points.push_back(i * section_size);
1021
1022    return split(tensor, split_points, axis);
1023  }

1027    size_t n_sections, size_t axis)

1046      split_points.push_back(pos);
1047    }
1048
1049    return split(tensor, split_points, axis);

1180  template <typename T>
1181  Tensor<T> Tensor<T>::reshape(const std::vector<ptrdiff_t>& new_shape) const
1182  {
1183    auto total = get_total_size();
1184
1185    // Resolve -1 dimension
1186    std::vector<size_t> resolved(new_shape.size());
1187    ptrdiff_t infer_idx = -1;
1188    size_t known_product = 1;
1189    for (size_t i = 0; i < new_shape.size(); ++i)
1190    {
1191      if (new_shape[i] == -1)
1192      {
1193        if (infer_idx >= 0)
1194          throw std::invalid_argument("Only one dimension can be inferred (-1)");
1195        infer_idx = static_cast<ptrdiff_t>(i);
1196      }
1197      else if (new_shape[i] < 0)
1198      {
1199        throw std::invalid_argument("Invalid dimension size: " + std::to_string(new_shape[i]));
1200      }
1201      else
1202      {
1203        resolved[i] = static_cast<size_t>(new_shape[i]);
1204        known_product *= resolved[i];
1205      }
1206    }
1207
1208    if (infer_idx >= 0)
1209    {
1210      if (known_product == 0 || total % known_product != 0)
1211        throw std::invalid_argument("Cannot infer dimension: total size " +
1212          std::to_string(total) + " is not divisible by " + std::to_string(known_product));
1213      resolved[infer_idx] = total / known_product;
1214    }
1215
1216    // Validate total size
1217    size_t new_total = 1;
1218    for (auto d : resolved) new_total *= d;
1219    if (new_total != total)
1220      throw std::invalid_argument("Cannot reshape tensor of size " +
1221        std::to_string(total) + " into shape " + shape_to_string(resolved));
1222
1223    // Use contiguous source (copy if needed)
1224    Tensor<T> src = m_isContiguous ? *this : copy();
1225
1226    Tensor<T> ret;
1227    ret.m_data = src.m_data;
1228    ret.m_offset = src.m_offset;
1229    ret.m_shape = resolved;
1230    ret.m_isContiguous = true;
1231    ret.m_viewType = ViewType::Base;
1232
1233    // Compute row-major strides
1234    ret.m_strides.resize(resolved.size());
1235    if (!resolved.empty())
1236    {
1237      ret.m_strides.back() = 1;
1238      for (auto i = static_cast<ptrdiff_t>(resolved.size()) - 2; i >= 0; --i)
1239        ret.m_strides[i] = ret.m_strides[i + 1] * static_cast<ptrdiff_t>(resolved[i + 1]);
1240    }
1241
1242    return ret;
1243  }
