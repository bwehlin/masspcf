345  /**
346   * Compute the broadcast-compatible output shape from two input shapes (NumPy rules).
347   *
348   * Shapes are compared right-to-left. Dimensions match if they are equal or one of them is 1.
349   * Missing leading dimensions are treated as size 1.
350   *
351   * @param a shape of the first operand
352   * @param b shape of the second operand
353   * @return the broadcast output shape
354   * @throws std::invalid_argument if the shapes are not broadcast-compatible
355   */
356  inline std::vector<size_t> broadcast_shapes(
357    const std::vector<size_t>& a,
358    const std::vector<size_t>& b)
359  {
360    size_t ndim = std::max(a.size(), b.size());
361    std::vector<size_t> result(ndim);
362
363    for (size_t i = 0; i < ndim; ++i)
364    {
365      size_t da = (i < ndim - a.size()) ? 1 : a[i - (ndim - a.size())];
366      size_t db = (i < ndim - b.size()) ? 1 : b[i - (ndim - b.size())];
367
368      if (da == db)
369        result[i] = da;
370      else if (da == 1)
371        result[i] = db;
372      else if (db == 1)
373        result[i] = da;
374      else
375        throw std::invalid_argument("Shapes are not broadcast-compatible: " +
376            shape_to_string(a) + " and " + shape_to_string(b));
377    }
378    return result;
379  }
