189    if (self == nullptr) {

191    }

300/// For multiple inheritance types we need to recursively register/deregister base pointers for any
301/// base classes with pointers that are difference from the instance value pointer so that we can
302/// correctly recognize an offset base class pointer. This calls a function with any offset base
303/// ptrs.
304inline void traverse_offset_bases(void *valueptr,
305                                  const detail::type_info *tinfo,
306                                  instance *self,
307                                  bool (*f)(void * /*parentptr*/, instance * /*self*/)) {
308    for (handle h : reinterpret_borrow<tuple>(tinfo->type->tp_bases)) {
309        if (auto *parent_tinfo = get_type_info(reinterpret_cast<PyTypeObject *>(h.ptr()))) {
310            for (auto &c : parent_tinfo->implicit_casts) {
311                if (c.first == tinfo->cpptype) {
312                    auto *parentptr = c.second(valueptr);
313                    if (parentptr != valueptr) {
314                        f(parentptr, self);
315                    }
316                    traverse_offset_bases(parentptr, parent_tinfo, self, f);
317                    break;
318                }
319            }
320        }
321    }
322}

356    assert(ptr);

453/// Clears all internal data from the instance and removes it from registered instances in
454/// preparation for deallocation.
455inline void clear_instance(PyObject *self) {
456    auto *instance = reinterpret_cast<detail::instance *>(self);
457
458    // Deallocate any values/holders, if present:
459    for (auto &v_h : values_and_holders(instance)) {
460        if (v_h) {
461
462            // We have to deregister before we call dealloc because, for virtual MI types, we still
463            // need to be able to get the parent pointers.
464            if (v_h.instance_registered()
465                && !deregister_instance(instance, v_h.value_ptr(), v_h.type)) {
466                pybind11_fail(
467                    "pybind11_object_dealloc(): Tried to deallocate unregistered instance!");
468            }
469
470            if (instance->owned || v_h.holder_constructed()) {
471                v_h.type->dealloc(v_h);
472            }
473        } else if (v_h.holder_constructed()) {
474            v_h.type->dealloc(v_h); // Disowned instance.
475        }
476    }
477    // Deallocate the value/holder layout internals:
478    instance->deallocate_layout();
479
480    if (instance->weakrefs) {
481        PyObject_ClearWeakRefs(self);
482    }
483
484    PyObject **dict_ptr = _PyObject_GetDictPtr(self);
485    if (dict_ptr) {
486        Py_CLEAR(*dict_ptr);
487    }
488
489    if (instance->has_patients) {
490        clear_patients(self);
491    }
492}

612/// buffer_protocol: Fill in the view as specified by flags.
613extern "C" inline int pybind11_getbuffer(PyObject *obj, Py_buffer *view, int flags) {
614    // Look for a `get_buffer` implementation in this type's info or any bases (following MRO).
615    type_info *tinfo = nullptr;
616    for (auto type : reinterpret_borrow<tuple>(Py_TYPE(obj)->tp_mro)) {
617        tinfo = get_type_info((PyTypeObject *) type.ptr());
618        if (tinfo && tinfo->get_buffer) {
619            break;
620        }
621    }
622    if (view == nullptr || !tinfo || !tinfo->get_buffer) {
623        if (view) {
624            view->obj = nullptr;
625        }
626        set_error(PyExc_BufferError, "pybind11_getbuffer(): Internal error");
627        return -1;
628    }
629    std::memset(view, 0, sizeof(Py_buffer));
630    std::unique_ptr<buffer_info> info = nullptr;
631    try {
632        info.reset(tinfo->get_buffer(obj, tinfo->get_buffer_data));
633    } catch (...) {
634        try_translate_exceptions();
635        raise_from(PyExc_BufferError, "Error getting buffer");
636        return -1;
637    }
638    if (info == nullptr) {
639        pybind11_fail("FATAL UNEXPECTED SITUATION: tinfo->get_buffer() returned nullptr.");
640    }
641
642    if ((flags & PyBUF_WRITABLE) == PyBUF_WRITABLE && info->readonly) {
643        // view->obj = nullptr;  // Was just memset to 0, so not necessary
644        set_error(PyExc_BufferError, "Writable buffer requested for readonly storage");
645        return -1;
646    }
647
648    // Fill in all the information, and then downgrade as requested by the caller, or raise an
649    // error if that's not possible.
650    view->itemsize = info->itemsize;
651    view->len = view->itemsize;
652    for (auto s : info->shape) {
653        view->len *= s;
654    }
655    view->ndim = static_cast<int>(info->ndim);
656    view->shape = info->shape.data();
657    view->strides = info->strides.data();
658    view->readonly = static_cast<int>(info->readonly);
659    if ((flags & PyBUF_FORMAT) == PyBUF_FORMAT) {
660        view->format = const_cast<char *>(info->format.c_str());
661    }
662
663    // Note, all contiguity flags imply PyBUF_STRIDES and lower.
664    if ((flags & PyBUF_C_CONTIGUOUS) == PyBUF_C_CONTIGUOUS) {
665        if (PyBuffer_IsContiguous(view, 'C') == 0) {
666            std::memset(view, 0, sizeof(Py_buffer));
667            set_error(PyExc_BufferError,
668                      "C-contiguous buffer requested for discontiguous storage");
669            return -1;
670        }
671    } else if ((flags & PyBUF_F_CONTIGUOUS) == PyBUF_F_CONTIGUOUS) {
672        if (PyBuffer_IsContiguous(view, 'F') == 0) {
673            std::memset(view, 0, sizeof(Py_buffer));
674            set_error(PyExc_BufferError,
675                      "Fortran-contiguous buffer requested for discontiguous storage");
676            return -1;
677        }
678    } else if ((flags & PyBUF_ANY_CONTIGUOUS) == PyBUF_ANY_CONTIGUOUS) {
679        if (PyBuffer_IsContiguous(view, 'A') == 0) {
680            std::memset(view, 0, sizeof(Py_buffer));
681            set_error(PyExc_BufferError, "Contiguous buffer requested for discontiguous storage");
682            return -1;
683        }
684
685    } else if ((flags & PyBUF_STRIDES) != PyBUF_STRIDES) {
686        // If no strides are requested, the buffer must be C-contiguous.
687        // https://docs.python.org/3/c-api/buffer.html#contiguity-requests
688        if (PyBuffer_IsContiguous(view, 'C') == 0) {
689            std::memset(view, 0, sizeof(Py_buffer));
690            set_error(PyExc_BufferError,
691                      "C-contiguous buffer requested for discontiguous storage");
692            return -1;
693        }
694
695        view->strides = nullptr;
696
697        // Since this is a contiguous buffer, it can also pretend to be 1D.
698        if ((flags & PyBUF_ND) != PyBUF_ND) {
699            view->shape = nullptr;
700            view->ndim = 0;
701        }
702    }
703
704    // Set these after all checks so they don't leak out into the caller, and can be automatically
705    // cleaned up on error.
706    view->buf = info->ptr;
707    view->internal = info.release();
708    view->obj = obj;
709    Py_INCREF(view->obj);
710    return 0;
711}

713/// buffer_protocol: Release the resources of the buffer.
714extern "C" inline void pybind11_releasebuffer(PyObject *, Py_buffer *view) {
715    delete (buffer_info *) view->internal;
716}
