22  Tensor<T>::Tensor(const std::vector<size_t>& shape, const T& init)
23    : m_shape(shape)
24  {
25    auto sz = get_total_size();
26    m_data = std::make_shared<T[]>(sz);
27    std::fill_n(m_data.get(), sz, init);
28
29    // Compute strides
30    if (!m_shape.empty())
31    {
32      m_strides = m_shape;
33
34      std::partial_sum(m_shape.rbegin(), std::prev(m_shape.rend()), std::next(m_strides.rbegin()), std::multiplies<>());
35      m_strides.back() = 1;
36    }
37  }

166  template <typename UnaryFunc>
167#ifndef __CUDACC__
168  requires std::invocable<UnaryFunc, std::vector<size_t>>
169#endif
170  void Tensor<T>::walk(UnaryFunc&& f) const
171  {
172    if (m_shape.empty() || std::any_of(m_shape.begin(), m_shape.end(), [](size_t n){ return n == 0; }))
173    {
174      return;
175    }
176    auto ndim = m_shape.size();
177
178    std::vector<size_t> cur(ndim, 0_uz);
179
180    while (true)
181    {
182      if constexpr (std::is_same_v<decltype(f(cur)), bool>)
183      {
184        // If f returns bool, stop walking on false being returned.
185        if (!f(cur))
186        {
187          return;
188        }
189      }
190      else
191      {
192        f(cur);
193      }
194
195
196      for (ptrdiff_t i = ndim - 1; i >= 0 ; --i)
197      {
198        ++cur[i];
199
200        if (cur[i] < m_shape[i])
201        {
202          break;
203        }
204
205        if (i == 0)
206        {
207          return;
208        }
209
210        cur[i] = 0;
211      }
212    }
213  }

260  template <typename SliceVector>
261  Tensor<T> Tensor<T>::extract(SliceVector sliceVector) const
262  {
263    Tensor ret;
264
265    ret.m_data = m_data; // view
266
267    // temporary just to get sizes
268    ret.m_shape = m_shape;
269    ret.m_strides = m_strides;
270
271    ret.m_offset = m_offset;
272
273    std::vector<size_t> dimsToDrop;
274
275    // For now, we'll drop the assumption that the tensor is contiguous in memory
276    // as soon as we extract a subtensor. There are, however, some cases where the resulting subtensor would be
277    // contiguous that we can try to optimize for in the future (e.g., extracting the top n rows of a matrix).
278    // Things will work fine with this assumption dropped but certain operations could be a little slower (probably
279    // unlikely to matter for the type of things we're targeting).
280    ret.m_isContiguous = false;
281
282    size_t i = 0;
283    for (auto & slice : sliceVector)
284    {
285      std::visit([i, &slice, &ret, &dimsToDrop](auto&& arg) {
286        using argT = std::decay_t<decltype(arg)>;
287        if constexpr (std::is_same_v<argT, SliceIndex>)
288        {
289          ret.m_shape[i] = 1;
290          ret.m_offset += arg.index * ret.m_strides[i];
291          dimsToDrop.emplace_back(i);
292        }
293        else if constexpr (std::is_same_v<argT, SliceRange>)
294        {
295          if (!arg.start)
296          {
297            arg.start = 0_z;
298          }
299          if (!arg.stop)
300          {
301            arg.stop = static_cast<ptrdiff_t>(ret.m_shape[i]);
302          }
303          if (!arg.step)
304          {
305            arg.step = 1_z;
306          }
307
308          auto start = *arg.start;
309          auto stop = *arg.stop;
310          auto step = *arg.step;
311
312          if (start < 0_z)
313            start = 0;
314          if (stop > static_cast<std::ptrdiff_t>(ret.m_shape[i]))
315            stop = static_cast<std::ptrdiff_t>(ret.m_shape[i]);
316
317          if (step == 0_z)
318          {
319            ret.m_shape[i] = 0;
320          }
321          else if (step > 0)
322          {
323            if (stop <= start)
324            {
325              ret.m_shape[i] = 0;
326            }
327            else
328            {
329              ret.m_shape[i] = (stop - start + step - 1_z) / step;
330            }
331          }
332          else
333          {
334            throw std::runtime_error("Negative step not supported...");
335          }
336
337          ret.m_offset += start * ret.m_strides[i]; // use clamped start
338          ret.m_strides[i] *= step;
339
340        }
341        // For SliceAll, don't modify shape
342      }, slice);
343      ++i;
344    }
345
346    size_t nDroppedDims = 0_uz;
347    for (auto dim : dimsToDrop)
348    {
349      ret.m_shape.erase(ret.m_shape.begin() + dim - nDroppedDims);
350      ret.m_strides.erase(ret.m_strides.begin() + dim - nDroppedDims);
351      ++nDroppedDims;
352    }
353
354    return ret;
355  }
